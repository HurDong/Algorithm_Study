# 프로그래머스 SQL : 저자별 카테고리 별 매출액 집계

SELECT B.AUTHOR_ID, A.AUTHOR_NAME, B.CATEGORY, SUM(S.SALES*B.PRICE) AS TOTAL_SALES
FROM BOOK B, AUTHOR A, BOOK_SALES S
WHERE B.BOOK_ID = S.BOOK_ID AND B.AUTHOR_ID = A.AUTHOR_ID
    AND S.SALES_DATE LIKE "2022-01-%"
GROUP BY B.AUTHOR_ID, B.CATEGORY
ORDER BY 1, 3 DESC
